# 用户信息接口

# 用户注册发送验证码

  • 请求方式GET
  • 请求地址http:10.7.178.116:3000/api/getMailCode
  • 请求参数(用于接受验证码的邮箱)
    params:{
    email=xxx@qq.com,
}
  • 返回参数
    • 成功
    status:0,
message:'发送验证码成功',
    • 失败
    status:1,
message:'发送失败',

# 注册

  • 请求方式POST
  • 请求地址http:10.7.178.116:3000/api/register
  • 请求参数
    body:{
    username='用户名字',
    password='用户密码',
    code='验证码',
}
  • 返回参数
    • 成功
    status:0,
message:'注册成功',
    • 失败
    status:1,
message:err(具体失败信息),

# 用户登录

  • 请求方式POST
  • 请求地址http:10.7.178.116:3000/api/signin
  • 请求参数(用于接受验证码的邮箱)
  body:{
  username='用户名字',
  password='用户密码',
  }
  • 返回参数
    • 成功（token后面身份验证标识）
    status:0,
message:'登录成功',
token:'Bearer '+tokenStr,
    • 失败
    status:1,
message:err(具体失败信息),
