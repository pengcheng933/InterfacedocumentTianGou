# 获取用户信息

  • 请求方式GET
  • 请求地址http:10.7.178.116:3000/my/user
  • 请求参数
    headers:{
   "Authorization":token
}
params:{
    name='商品名称',
}
  • 返回参数
    • 成功
      status:0,
message:'获取成功',
data: {
    username,
    avatarUrl,
},
    • 失败
      status:1,
message:err,

# 更改用户头像

  • 请求方式POST
  • 请求地址http:10.7.178.116:3000/my/ava
  • 请求参数(名字，密码，头像地址)
    params:{
  name,
  password,
  avatarUrl,
}
  • 返回参数
    • 成功
      status:0,
                message:'修改成功',
    • 失败
      status:1,
message:'密码错误',

      status:1,
message:'用户名不存在',
